⚡ SETUP SUPABASE
Lần đầu sử dụng, bạn cần chạy SQL này trong Supabase Dashboard → SQL Editor:
-- Chạy trong Supabase SQL Editor create table if not exists exam_sessions ( id text primary key, mode text, current_page int default 1, pages jsonb default '[]', page_states jsonb default '{}', tool text default 'pen', color text default '#ff4d6d', size int default 3, updated_at timestamptz default now() ); create table if not exists exam_bg_images ( id bigint generated always as identity primary key, session_id text references exam_sessions(id) on delete cascade, page_num int not null, data_url text not null, saved_at timestamptz default now(), unique(session_id, page_num) ); create table if not exists exam_gallery ( id bigint generated always as identity primary key, session_id text references exam_sessions(id) on delete cascade, page_num int not null, data_url text not null, label text, saved_at timestamptz default now(), unique(session_id, page_num) ); alter table exam_sessions enable row level security; alter table exam_bg_images enable row level security; alter table exam_gallery enable row level security; create policy "allow all" on exam_sessions for all using (true) with check (true); create policy "allow all" on exam_bg_images for all using (true) with check (true); create policy "allow all" on exam_gallery for all using (true) with check (true);
